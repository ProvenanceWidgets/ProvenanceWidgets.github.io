# Radio Button

A Radio Button is a user interface control element that allows users to select one option from a set of options.

## Playground

```jsx live
<ProvenanceRadiobutton 
  id="jsx-provenance-radiobutton"
  name="jsx-provenance-radiobutton"
  data={[
    { label: 'New York', value: 'New York' },
    { label: 'Rome', value: 'Rome' },
    { label: 'London', value: 'London' },
    { label: 'Istanbul', value: 'Istanbul' },
    { label: 'Paris', value: 'Paris' }
  ]}
  selected={'New York'}
  freeze={false}
  visualize={true}
  provenanceChange={console.log} // hardcoded
  selectedChange={console.log} // hardcoded
/>
```

        ```typescript
        import { RadiobuttonComponent } from 'provenance-widgets';
        import type { NgElement, WithProperties } from '@angular/elements'

        // Create a new instance of the Radiobutton group
        const radiobutton = document.createElement("web-provenance-radiobutton") as NgElement & WithProperties<RadiobuttonComponent>;

        const data = [
            { label: 'New York', value: 'New York' },
            { label: 'Rome', value: 'Rome' },
            { label: 'London', value: 'London' },
            { label: 'Istanbul', value: 'Istanbul' },
            { label: 'Paris', value: 'Paris' }
        ];
        // Set the properties
        radiobutton.data = data;
        radiobutton.selected = 'New York';
        radiobutton.freeze = false;
        radiobutton.visualize = true;
        
        radiobutton.addEventListener('provenanceChange', (event) => {
            console.log(event.detail);
        });

        radiobutton.addEventListener('selectedChange', (event) => {
            console.log(event.detail);
            // Mandatory to update the selected property manually
            radiobutton.selected = event.detail;
        });
        
        // Append it to the DOM
        document.body.appendChild(radiobutton);
        ```
